	{\Huge \textbf{KẾT THÚC PHẦN 1}} \\[1.5cm]
	
	\rule{0.8\textwidth}{0.4pt} \\[0.8cm]
	
	\parbox{0.9\textwidth}{
		\centering
		\textit{
			Phần 1 của giáo trình đã cung cấp cho bạn một nền tảng lý thuyết toàn diện, 
			từ những nguyên lý xử lý ảnh cơ bản, các kỹ thuật trích xuất đặc trưng cổ điển, 
			các kiến trúc mạng nơ-ron kinh điển (CNNs), cho đến sự xuất hiện của Transformer 
			trong thị giác máy tính và các mô hình tiên tiến cho học biểu diễn, 
			sinh ảnh/video và thị giác đa phương thức. Với nền tảng này, bạn đã sẵn sàng 
			để bước sang Phần 2, nơi chúng ta sẽ biến những lý thuyết này thành các kỹ năng 
			và quy trình thực chiến để xây dựng các ứng dụng CV trong thực tế.
		}
	} \\[1cm]
	
	\rule{0.8\textwidth}{0.4pt} \\[2cm]
\end{center}

\noindent
\textbf{Tóm lược nội dung Phần 1:}
\begin{itemize}
	\item \textbf{Chương 1 – Nhập môn và Nguyên lý nền tảng:} 
	Định nghĩa CV, nền tảng toán học và xử lý tín hiệu, biểu diễn ảnh/video số, 
	cảm biến và computational imaging, cùng các vấn đề đạo đức trong AI thị giác.
	\item \textbf{Chương 2 – Xử lý ảnh nền tảng và đặc trưng cổ điển:} 
	Các phép toán ảnh cơ bản, phát hiện biên, đặc trưng keypoint (Harris, FAST), 
	các mô tả đặc trưng thủ công (SIFT, SURF, ORB, HOG), biểu diễn BoVW/Fisher/VLAD, 
	và hạn chế của các phương pháp truyền thống.
	\item \textbf{Chương 3 – Kỷ nguyên CNNs:} 
	Các thành phần cơ bản (convolution, pooling, activation), 
	kiến trúc kinh điển (AlexNet, VGG), hiện đại (ResNet, DenseNet, EfficientNet), 
	kỹ thuật tối ưu (BN, Dropout, Data Augmentation), và CNN nhỏ gọn cho Edge AI.
	\item \textbf{Chương 4 – Kỷ nguyên Transformer trong thị giác:} 
	Vision Transformer (ViT), patch embedding, self-attention cho ảnh, 
	các biến thể (Swin, DeiT), hybrid CNN+Transformer, và so sánh với CNN.
	\item \textbf{Chương 5 – Các bài toán CV cốt lõi và kiến trúc chuyên biệt:} 
	Object Detection (YOLO, Faster R-CNN, DETR), 
	Image Segmentation (U-Net, Mask R-CNN, SAM, SEEM), 
	pose estimation, tracking hiện đại, domain adaptation, 
	và foundation models cho vision tasks.
	\item \textbf{Chương 6 – Học biểu diễn và Kỷ nguyên dữ liệu:} 
	Self-supervised learning (SimCLR, MoCo, MAE), deep metric learning, 
	data-centric AI, zero/few-shot, continual learning, và foundation models (DINOv2, OpenCLIP).
	\item \textbf{Chương 7 – Các mô hình sinh và sáng tạo thị giác:} 
	GANs (DCGAN, StyleGAN), diffusion models (DDPM, Stable Diffusion), 
	ControlNet, text-to-video (Sora, Pika Labs), neural rendering, 
	và mô hình đa phương thức sinh ảnh/video.
	\item \textbf{Chương 8 – Thị giác đa phương thức và 3D:} 
	CLIP, LVLMs (GPT-4o, LLaVA, Gemini), NeRFs, Gaussian Splatting, 
	AR/VR, robotics, video-language models, và generative AI 3D+4D.
\end{itemize}
